package io.github.guilhebl.util;

import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class RegexUtil {

    /**
   ^                 # start-of-string
   (?=.*[0-9])       # a digit must occur at least once
   (?=.*[a-z])       # a lower case letter must occur at least once
   (?=.*[A-Z])       # an upper case letter must occur at least once
   
   ### (?=.*[@#$%^&+=])  # a special character must occur at least once
   
   (?=[^\s]+$)       # no whitespace allowed in the entire string
   .{6,}             # anything, at least eight places though
   $                 # end-of-string
   */
    public static final String PASSWORD_PATTERN = "^(?=.*[0-9])(?=.*[a-z])(?=[^\\s]+$).{6,}$";

    private static final String EMAIL_PATTERN = "^[_A-Za-z0-9-]+(\\." +
            "[_A-Za-z0-9-]+)*@[A-Za-z0-9]+(\\.[A-Za-z0-9]+)*" +
            "(\\.[A-Za-z]{2,})$";
    
    private static final String FILE_NAME_PATTERN = "[_a-zA-Z0-9\\-\\.]+"; 
    
    private static final String PHONE_PATTERN = "^[+]?([0-9]*[\\.\\s\\-\\(\\)]|[0-9]+){3,24}$";
    
    private static RegexUtil instance;
    private Pattern patternPassword;
    private Pattern patternEmail;
    private Pattern patternFileName;
    private Pattern patternPhone;
    private Matcher matcher; 
    
    private RegexUtil() {
        patternEmail = Pattern.compile(EMAIL_PATTERN);
        patternFileName = Pattern.compile(FILE_NAME_PATTERN);
        patternPhone = Pattern.compile(PHONE_PATTERN);
        patternPassword = Pattern.compile(PASSWORD_PATTERN);
    }
    
    public static RegexUtil getInstance() {
        if (instance == null) 
        {
            instance = new RegexUtil();
        }
        return instance;
    }

    public boolean validateIntegerID(Integer id) {
        if (id != null && id >= 0) {          
            return true;
        }     
        return false;
    }

    public boolean validateStringTypeRegex(String text, String patternType) {
        if (text != null && !text.equals("")) {   
            
            if (patternType.equals("email")) {
                matcher = patternEmail.matcher(text);  
            } else if (patternType.equals("password")) {
                matcher = patternPassword.matcher(text);   
            } else if (patternType.equals("phone")) {
                matcher = patternPhone.matcher(text);  
            } else if (patternType.equals("fileName")) {
                matcher = patternFileName.matcher(text);   
            }
            
            if(matcher.matches()){
                return true;
            }
        }     
        return false;
    }
    
    public boolean validateEmail(String email) {
        return validateStringTypeRegex(email, "email");
    }
    
    public boolean validateFileName(String fileName) {
        return validateStringTypeRegex(fileName, "fileName");
    }
    
    public boolean validatePhoneNumber(String phone) {
        return validateStringTypeRegex(phone, "phone");
    }
}
